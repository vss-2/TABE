function fibonacci(n)
	local val_antes = -1
	local val_atual =  1
	local contador  = -1
	return function()
		fib_atual = val_antes + val_atual
		if contador+1 > n then
			return nil
		else
			val_antes = val_atual
			val_atual = fib_atual
			contador  = contador + 1
			return fib_atual
		end
	end
end

s = 0
for i in fibonacci(5) do
	s = s + i
end
print(s)
